//*********************__FULL__EXAMPLE__**********************************

services:
  db:
    image: postgres:15
    container_name: pg_db   # Same as --name in Dokerfile. Used for Docker DNS system to locate this container.
    hostname: pg_db_host    # By default = short container ID. This name is used only inside THIS container(for logs for example).
    platform: linux/amd64   # Tag used mostly for Dockers on Mac platform
    pull_policy: always     # Strategy for taking image from repo.
    restart: always         # Strategy for Docker restart. 
    ports:                  # Port forwarding, 5432 = default PG port
      - "5432:5432"         # port better to write in " to prevent reading number as 60-base system number
      - "5000-5010:5000"    # use first available port from the mentioned range. Used with "scale" option (see below)
    expose:                 # Wich ports exposed outside from Docker container
      - "5432"
    environment:            # Inject settings into container. Keep data in docker-compose file, so it's NOT for secrets
      POSTGRES_PASSWORD: secret
      POSTGRES_USER: admin
      POSTGRES_DB: mydb
    env_file:               # Inject settings into container. Keep data in outside file, so it's GOOD for secrets
      - .env                # The file should be added into .gitignore not nopt save by mistake your secrets
    volumes:                # Create volumes
      - pgdata:/var/lib/postgresql/data
      - ./init:/docker-entrypoint-initdb.d:ro
      - type: tmpfs         # special type of volume (not bind or volume)
        target: /tmp        # where in container save this volume
        tmpfs:
          size: 100m        # memory size for this volume
          mode: 1777        # credentials of the volume
    networks:               # Create networks the container will be connected to. If not mentioned - the default network will be used.
      - backend
      - backend_2
    healthcheck:            # The same as HEALTCHECK in Dockerfile
      test: ["CMD-SHELL", "pg_isready -U admin"]
      interval: 10s         # Do test each interval
      timeout: 5s           # Wait for test response during timeout
      retries: 5            # Tries count before failure
      start_period: 30s     # first test will be executed after this time starting from docker creatioon
    deploy:                 # SWARM section only
      resources:
        limits:
          cpus: "0.5"
          memory: 512m
        reservations:
          cpus: "0.25"
          memory: 256m
    logging:
      driver: json-file     # MAX stuping and ambiguious name - not related at all to JSON format!!! Logs kept on host.
      options:
        max-size: "10m"             # Max size for each log
        max-file: "3"               # Max number of files to be kept. After last will overwrite the first.
        labels: "env,app"           # The labels will be added per each record inside the log
        env: "APP_VERSION,APP_ENV"  # The env variables will be added per each record inside the log
        tag: "{{.ImageName}}/{{.Name}}/{{.ID}}"    # Template for container identification inside the log
        compress: "true"            # What to do once current file reach his max-size
    labels:                 # same as LABEL in Dockerfile
      env: prod
      app: postgres
    user: "1000:1000"       # same as USER in Dockerfile
    ulimits:                # Linux configuration for container. Can impact on HOST.
      nofile:
        soft: 1024
        hard: 2048
    stop_signal: SIGTERM    # same as STOPSIG in Dockerfile
    stop_grace_period: 30s  # time before hardcode kill signal come
    init: true              # start lightweight process to take care about zombie process inside the container
    privileged: false       # grant all HOST privileges for container (not a good practice, for best - see next 2 options)
    cap_drop:               # remove all HOST privileges from container (there are some - ~15 - that exists by default)
      - ALL
    cap_add:                # add only the following privileges of HOST to container (there are ~50 options)
      - CHOWN
      - SETUID
      - SETGID
    security_opt:           # security options
      - no-new-privileges:true  # block all privileges elevations inside the container (in Linux there are 2 options for it)
      - seccomp:profile.json    # very tinny configuration for Linux core access
    sysctls:                # Linux configuration for container 
      net.core.somaxconn: 1024
    dns:                    # additional DNS servers, usually not in use
      - 8.8.8.8
      - 1.1.1.1
    dns_search:     # postfix used to resolve full name of service in DNS table (Not jus "db", but "db.mycompany.com" will be searched)
      - mycompany.com
    shm_size: 256m          # size of default shared memory (not tmpfs!!!, mounted by OS). Default = 64m, recommended >= 256m
    ipc: shareable          # connection between containers like network, but in RAM. Faster than network in ~1000 times.
    pid: host               # allow to see all the processes on HOST from container. Can send SIGNALS to HOST from container!!!
    profiles:               # allow to start services with specified profiles only. If not mentioned - service will start always.
      - production          # name of profile. Container started with command "docker compose --profile production up"
    scale: 2                # Start N copies of service without SWARM. Can't use with container_name as it's unique.

  app:
    build:                          # how exactly to build the service
      context: ./app                # where Dockerfile sits
      dockerfile: Dockerfile.prod   # if Dockerfile has a non-standard name
      args:                         # same as ARGS in Dockerfile
        APP_VERSION: 1.0
        ENV: production
    container_name: myapp
    hostname: myapp_host
    pull_policy: build      # always build the container from scratch. Never pulled from registry
    restart: on-failure
    ports:
      - "8000:8000"
    expose:
      - "8000"
      - "9000"
    env_file:
      - .env
    environment:
      DB_HOST: db
      DB_PORT: 5432
    volumes:
      - ./app:/app
      - staticfiles:/app/static      
    networks:
      - backend
      - frontend
    depends_on:                     # This container will be started ONLY after mentioned inside containers will be UP.
      db:
        condition: service_healthy  # Wait for mentiond container(db) healthcheck return true.
    stdin_open: true                # same as "-i" in "docker run -it" command. Used for manual inspection inside the container
    tty: true                       # same as "-t" in "docker run -it" command. Used for manual inspection inside the container
    working_dir: /app               # same as WORKDIR in Dockerfile. 
    command: python manage.py runserver 0.0.0.0:8000 # same as CMD in Dockerfile
    entrypoint: ["sh", "-c"]        # same as ENTRYPOINT in Dockerfile
    extra_hosts:                    # allow connection from container to HOST (automatically for Mac/Win, but not for Linux)
      - "host.docker.internal:host-gateway"
    stop_signal: SIGINT
    stop_grace_period: 10s
    init: true
    privileged: false
    cap_drop:
      - ALL
    cap_add:
      - NET_BIND_SERVICE
    security_opt:
      - no-new-privileges:true
      - seccomp:profile.json
    sysctls:
      net.ipv4.tcp_syncookies: 1
    dns:
      - 8.8.8.8
    shm_size: 128m
    profiles:
      - production
      - debug
    scale: 2

  nginx:
    image: nginx:latest
    container_name: mynginx
    hostname: mynginx_host
    platform: linux/amd64
    pull_policy: missing        #default behavior - try to find locally, if not exists - try to load from remote registry 
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    expose:
      - "80"
      - "443"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf:ro
      - staticfiles:/usr/share/nginx/html:ro
      - type: bind
        source: ./certs
        target: /etc/nginx/certs
        read_only: true
    networks:
      - frontend
    depends_on:
      - app
    logging:
      driver: syslog
      options:
        syslog-address: "tcp://localhost:514"
    stop_signal: SIGQUIT
    stop_grace_period: 30s
    cap_drop:
      - ALL
    cap_add:
      - NET_BIND_SERVICE
    security_opt:
      - no-new-privileges:true

  redis:
    image: redis:7
    container_name: myredis
    hostname: myredis_host
    platform: linux/amd64
    pull_policy: always
    restart: always
    ports:
      - "6379:6379"
    expose:
      - "6379"
    volumes:
      - redisdata:/data
    networks:
      - backend
    command: redis-server --appendonly yes --requirepass secret
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 5s
      timeout: 3s
      retries: 3
    stop_signal: SIGTERM
    stop_grace_period: 10s
    sysctls:
      net.core.somaxconn: 512
    ulimits:
      nofile:
        soft: 65536
        hard: 65536


  worker:
    build: ./worker
    container_name: myworker
    hostname: myworker_host
    restart: on-failure
    env_file:
      - .env
    volumes:
      - ./worker:/app
    networks:
      - backend
    depends_on:
      - redis
      - db
    deploy:
      replicas: 3
      restart_policy:
        condition: on-failure
        delay: 5s
        max_attempts: 3
    init: true
    cap_drop:
      - ALL
    security_opt:
      - no-new-privileges:true
    profiles:
      - production
      - debug


volumes:                # A must have declaration of the volumes
  pgdata:
    driver: local
    labels:
      env: prod
      app: postgres
  staticfiles:
    driver: local
  redisdata:
    driver: local
    driver_opts:
      type: tmpfs
      device: tmpfs
      o: size=100m

networks:               # A must have declaration of the networks 
  backend:
    driver: bridge
    ipam:
      config:
        - subnet: 172.20.0.0/16
          gateway: 172.20.0.1
    labels:
      env: prod
  frontend:
    driver: bridge
    internal: false
  external_net:
    external: true
    name: existing_network

//************************__START__********************************

docker compose up                   # start all
docker compose up -d db             # start specific service
docker compose up --build           # start with build
docker compose up --force-recreate  # start with force build
docker compose -f com.prod.yml up   # start using specific YAML file
docker compose --profile prod up    # start with profile
